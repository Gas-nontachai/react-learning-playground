## Example API route
```ts
// src/app/api/students/route.ts
import {NextResponse} from 'next/server';
import {students} from '@/lib/data';

export async function GET() {
  return NextResponse.json(students);
}
```

### Example client component
```tsx
'use client';
import useSWR from 'swr';

const fetcher = (url: string) => fetch(url).then((res) => res.json());

export function StudentsList() {
  const {data, error, isLoading} = useSWR('/api/students', fetcher);

  if (isLoading) return <p>Loading...</p>;
  if (error) return <p>Failed to load students</p>;

  return (
    <ul className="space-y-3">
      {data.map((student: any) => (
        <li key={student.id} className="rounded-xl border border-slate-200 p-4">
          <div className="flex items-center justify-between">
            <p className="font-medium">{student.name}</p>
            <span className="text-sm text-slate-500">{student.progress}%</span>
          </div>
        </li>
      ))}
    </ul>
  );
}
```
