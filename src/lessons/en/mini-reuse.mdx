## Example: stat widget

```tsx
type StatProps = {
  label: string;
  value: string;
  trend?: 'up' | 'down';
};

export function Stat({label, value, trend}: StatProps) {
  const trendColor = trend === 'up' ? 'text-emerald-600' : 'text-rose-600';
  const trendIcon = trend === 'up' ? 'â–²' : 'â–¼';

  return (
    <div className="rounded-2xl border border-slate-200 bg-white p-4 shadow">
      <p className="text-sm text-slate-500">{label}</p>
      <p className="text-3xl font-semibold">{value}</p>
      {trend && <p className={`text-sm ${trendColor}`}>{trendIcon} 12% vs last week</p>}
    </div>
  );
}
```

Reuse this widget anywhere by passing different props.
