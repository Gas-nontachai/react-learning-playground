# 1.6 Events

Events let components react to user actionsâ€”clicks, key presses, form submissions, etc. In React you attach handlers directly to JSX elements.

<br />

## Event handler pattern
```jsx
function Button() {
  const handleClick = () => {
    console.log('Clicked!');
  };

  return <button onClick={handleClick}>Click</button>;
}
```
- Handler functions live inside the component so they can reach state/props
- React normalizes events so they behave similarly across browsers

<br />

## Example: todo toggles

```jsx
import {useState} from 'react';

type Todo = {
  id: number;
  text: string;
  done: boolean;
};

export default function TodoList() {
  const [todos, setTodos] = useState<Todo[]>([
    {id: 1, text: 'Review JSX', done: false},
    {id: 2, text: 'Practice props', done: true}
  ]);

  const toggleTodo = (id: number) => {
    setTodos((items) =>
      items.map((item) =>
        item.id === id ? {...item, done: !item.done} : item
      )
    );
  };

  return (
    <ul className="space-y-2">
      {todos.map((todo) => (
        <li
          key={todo.id}
          className="flex items-center gap-3 rounded border border-slate-200 p-3"
        >
          <button
            onClick={() => toggleTodo(todo.id)}
            className={[
              'size-6 rounded-full border',
              todo.done
                ? 'border-emerald-600 bg-emerald-500'
                : 'border-slate-400'
            ].join(' ')}
          />
          <span className={todo.done ? 'text-slate-400 line-through' : ''}>
            {todo.text}
          </span>
        </li>
      ))}
    </ul>
  );
}
```

<br />

## Tips
- Prevent default browser behavior by calling `event.preventDefault()` inside the handler
- Prefer descriptive handler names: `handleSubmit`, `handleFilterChange`
- Inline arrow functions are fine for simple cases (like `onClick={() => toggle(id)}`); extract them when logic grows
